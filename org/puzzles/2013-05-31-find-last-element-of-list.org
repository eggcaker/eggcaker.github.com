#+BEGIN_EXAMPLE
myLast [x] = x
myLast (_:xs) = myLast xs
myLast' = foldr1 (const id)

myLast'' = foldr1 (flip const)
 
myLast''' = head . reverse
 
myLast'''' = foldl1 (curry snd)
 
myLast''''' [] = error "No end for empty lists!"  
myLast''''' x = x !! (length x -1)
